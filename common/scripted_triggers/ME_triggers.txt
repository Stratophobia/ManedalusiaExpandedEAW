MDL_has_set_path_rules = {
  custom_trigger_tooltip = {
    tooltip = ME_has_set_path_rules_tt
    hidden_trigger = {
      OR = {
        has_game_rule = {
					rule = MDL_ai_behavior
					option = MDL_VELVET
				}
        has_game_rule = {
					rule = MDL_ai_behavior
					option = MDL_CRIMSON_SULTAN
				}
        has_game_rule = {
					rule = MDL_ai_behavior
					option = MDL_SENATE
				}
        has_game_rule = {
					rule = MDL_ai_behavior
					option = MDL_ULTRAMAR
				}
        OR = {
          MDL = { has_country_flag = MDL_AI_CRIMSON_SULTAN }
          MDL = { has_country_flag = MDL_AI_SENATE }
          MDL = { has_country_flag = MDL_AI_VELVET }
          MDL = { has_country_flag = MDL_AI_ULTRAMAR }
        }
      }
    }
  }
}

	custom_trigger_tooltip = {
		tooltip = owns_any_cores_original_subject_tt
		hidden_trigger = {
			any_of_scopes = {
				array = original_cores
				owner = {
					OR = {
						tag = PREV
						is_subject_of = PREV
					}
				}
			}
		}
	}
}
