any_original_cores_owned_by_PREV_or_subject = {
	custom_trigger_tooltip = {
		tooltip = owns_any_cores_original_subject_tt
		hidden_trigger = {
			any_of_scopes = {
				array = original_cores
				owner = {
					OR = {
						tag = PREV
						is_subject_of = PREV
					}
				}
			}
		}
	}
}
